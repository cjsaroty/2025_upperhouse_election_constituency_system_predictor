import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Excel読み込み
df = pd.read_excel("survey_results.xlsx")

# 当落を数値化（当選=1, 落選=0）
df["当落"] = df["当落"].map({"当選": 1, "落選": 0})

# 3つのQ1列をまとめて分析用に整形
# → 各候補者がどのテーマを選んだかをワンホット化する
topics = ['景気・雇用','財政','消費税','教育','子育て支援','働き方','年金・医療・介護',
          '憲法','エネルギー・環境','外交・安全保障','治安','地方活性化・人口減',
          '防災・復興','農林水産業','政治とカネ','選挙制度','情報公開','選択的夫婦別姓',
          '性的少数者に関する政策','その他']

# 各テーマの列を0/1で作成
for t in topics:
    df[t] = ((df["Q1_1"] == t) | (df["Q1_2"] == t) | (df["Q1_3"] == t)).astype(int)

# 相関行列を作成
corr = df[topics + ["当落"]].corr()["当落"].sort_values(ascending=False)

# 結果表示
print("各テーマと当落との相関係数:")
print(corr)

# 可視化
plt.figure(figsize=(6,8))
sns.barplot(x=corr.values, y=corr.index, palette="coolwarm")
